INSERT INTO trip_metadata
(trip_id, driver_id, order_id, start_time,
end_time, start_location, destination,
total_distance, total_duration, average_speed,
max_speed, speeding_violations, trip_cost, status)
VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?);

Pattern: UPSERT (INSERT = UPDATE nếu key tồn tại)

Calculate: Distance, duration, speed, cost
